                        margin: 30
                    }
                }
            });

            var $indicators = $('.carousel-indicators').find('button[data-bs-target="#'+$carousel.attr('id')+'"]');

            $indicators.click(function() {
                $carousel.trigger('to.owl.carousel', $(this).attr('data-bs-slide-to'));
            });

            setTimeout(function() {
                var i = 0;
                $carousel.find('.owl-item').removeClass('impressed');
                $carousel.find('.owl-item').each(function() {
                    if ($(this).hasClass('active')) {
                        if (i == 0) $(this).addClass('impressed');
                        i++;
                    }
                });
            }, 200);

            $carousel.on('changed.owl.carousel', function(e) {
                setTimeout(function() {
                    var i = 0;
                    $carousel.find('.owl-item').removeClass('impressed');
                    $carousel.find('.owl-item').each(function() {
                        if ($(this).hasClass('active')) {
                            if (i == 0) {
                                $(this).addClass('impressed');
                                var currentIndex = $(this).find('.item').attr('slideIndex');
                                $indicators.each(function() {
                                    $(this).removeClass('active');
                                    if ( $(this).attr('data-bs-slide-to') ==  currentIndex) {
                                        $(this).addClass('active');
                                    }
                                });
                            }
                            i++;
                        }
                    });
                }, 200);
            });
        });
    });
});
</script>
